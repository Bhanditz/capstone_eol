6 def create_if_not_exists(klass, attributes)
 
7   found = nil
 
8   begin
 
9     searchable_attributes = {}
 
10     attributes.keys.each do |key|
 
11       # Specified ids could be stored as Fixnum, not just int:
 
12       if attributes[key].class == String or attributes[key].class == Integer or attributes[key].class == Fixnum or
 
13          attributes[key].class == TrueClass or attributes[key].class == FalseClass
 
14          if klass.columns.map {|c| c.name.to_sym}.include? key
 
15            searchable_attributes[key] = attributes[key]
 
16          end
 
17       elsif attributes[key].class != Array
 
18         key_id = "#{key}_id"
 
19         key_id = 'toc_id' if key_id == 'toc_item_id'
 
20         searchable_attributes[key_id] = attributes[key].id
 
21       end
 
22     end
 
23     # Assumes that .keys returns in same order as .values, which is appears is true:
 
24     begin
 
25       found = klass.send("find_by_" << searchable_attributes.keys.join('_and_'), searchable_attributes.values) unless
 
26         searchable_attributes.keys.blank?
 
27     rescue NoMethodError
 
28       raise "It seems there is a bad column on #{klass}. One of its expected attributes seems to be missing: " +
 
29             "#{searchable_attributes.keys.inspect}"
 
30     end
 
31     found = klass.send(:gen, attributes) if found.nil?
 
32   rescue ActiveRecord::RecordInvalid => e
 
33     puts "** Invalid Record : #{e.message}"
 
34   rescue ActiveRecord::StatementInvalid => e
 
35     raise e unless klass.name == "AgentsResource" # Okay, TOTAL hack.  But for some reason, this was the ONLY class
 
36                                                   # For which the above find() method failed.  No clue why.  - JRice
 
37   end
 
38   return found
 
39 end
