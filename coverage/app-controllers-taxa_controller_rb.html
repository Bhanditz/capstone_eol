416   def curators
 
417     # if this is named taxon_concept then the RSS feeds will be added to the page
 
418     # in Firefox those feeds are evaluated when the pages loads, so this should save some queries
 
419     @concept = find_taxon_concept
 
420     return if taxon_concept_invalid?(@concept)
 
421     @page_title = "Curators of #{@concept.title(@session_hierarchy)}"
 
422     @curators = @concept.curators
 
423     @curators.sort! do |a, b|
 
424       if a.family_name.strip.blank? && b.family_name.strip.blank? # last names blank, sort by first
 
425         a.given_name.strip <=> b.given_name.strip
 
426       elsif a.family_name.strip.blank?                            # A.last blank, sort A.first to B.last
 
427         a.given_name.strip <=> b.family_name.strip
 
428       elsif b.family_name.strip.blank?                            # B last name blank, sort B.first to A.last
 
429         a.family_name.strip <=> b.given_name.strip
 
430       elsif a.family_name.strip == b.family_name.strip            # sort last
 
431         a.given_name.strip <=> b.given_name.strip
 
432       else                                                        # then sort first
 
433         a.family_name.strip <=> b.family_name.strip
 
434       end
 
435     end

687   def get_suggested_search_results(querystring)
 
688     pluralized = querystring.pluralize
 
689     singular   = querystring.singularize
 
690     suggested_results_original = SearchSuggestion.find_all_by_term_and_active(singular, true, :order => 'sort_order') +
 
691                                  SearchSuggestion.find_all_by_term_and_active(pluralized, true, :order => 'sort_order')
 
692     
693     # bacteria has a singular bacterium and a plural bacterias so we need to search on the original term too
 
694     if querystring != pluralized && querystring != singular
 
695       suggested_results_original += SearchSuggestion.find_all_by_term_and_active(querystring, true, :order => 'sort_order')
 
696     end
 
697     
698     return [] if suggested_results_original.blank?
 
699     suggested_results_query = suggested_results_original.select {|i| i.taxon_id.to_i > 0}.map {|i| 'taxon_concept_id:' + i.taxon_id}.join(' OR ')
 
700     suggested_results_query = suggested_results_query.blank? ? "taxon_concept_id:0" : "(#{suggested_results_query})"
 
701     suggested_results  = TaxonConcept.search_with_pagination(suggested_results_query, params)
 
702     suggested_results_original = suggested_results_original.inject({}) {|res, sugg_search| res[sugg_search.taxon_id] = sugg_search; res}
 
703     suggested_results.each do |res| 
704       common_name = suggested_results_original[res['taxon_concept_id'][0].to_s].common_name 
705       res['common_name'] = [common_name]
 
706       res['preferred_common_name'] = common_name
 
707     end
 
708     suggested_results
 
709   end
