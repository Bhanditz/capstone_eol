126   def index
 
127     begin
 
128       @taxon_concept = TaxonConcept.find params[:taxon_concept_id] if params[:taxon_concept_id]
 
129     rescue ActiveRecord::RecordNotFound
 
130       render :text => "Don't know how to render #{ params.inspect }"
 
131       return
 
132     end
 
133     per_page = params[:per_page].to_i
 
134     per_page = 10 if per_page < 1
 
135     per_page = 50 if per_page > 50
 
136     page     = params[:page].to_i
 
137     page     = 1 if page < 1
 
138     case request.path
 
139     when /images/
 
140       respond_to do |format|
 
141         format.xml do
 
142           xml = $CACHE.fetch("taxa/#{@taxon_concept.id}/images/#{page}.#{per_page}/xml", :expires_in => 4.hours) do
 
143             images = @taxon_concept.images
 
144             {
 
145               :images           => images.paginate(:per_page => per_page, :page => page),
 
146               'num-images'      => images.length,
