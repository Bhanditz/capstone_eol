505   def random_pronouncable_password(size = 4)
 
506     v = %w(a e i o u y)
 
507     c = ('a'..'z').to_a - v - ['q'] + %w(qu ch cr fr nd ng nk nt ph pr rd sh sl sp st th tr)
 
508     (1..size).map{[c.rand, v.rand]}.flatten.map{|x| [x,x.upcase].rand}.join('')
 
509   end
